7 — True SPR for verification —

# The true SPR is calculated by running the simulation with the known parameters
# (FM = 1.0). Since the data were simulated with these exact parameters, comparing the estimated SPR and F/M to these true values tells us how well the model recovers the input parameters. Small differences are expected due to multinomial sampling.
simGTG   <- GTGDomeLBSPRSim2(LifeHistoryObj, FleetPars, SizeBins)
true_SPR <- simGTG$SPR
cat("True SPR:", round(true_SPR, 4), "\n")
## True SPR: 0.7258
cat("True F/M: 1.000\n")
## True F/M: 1.000

7.0.0.1 Step 2. Create the lengthComp object

The LengthComp object is the standard container for catch-at-length data. The only slot that changes between examples is dataType — set it to "Frequency" for frequency data or "Length" for individual measurements. The observationGroup slot should always be set to "Catch".

# This is shown for illustration — each example below creates its own
# lengthComp object after loading its specific dataset.
lengthComp <- new("LengthComp")
lengthComp@dataType         <- "Frequency"  # "Frequency" or "Length"
lengthComp@L_source         <- "FD"         # "FD" = fishery dependent
lengthComp@L_units          <- "cm"
lengthComp@L_type           <- "TL"         # TL = total length
lengthComp@header           <- TRUE
lengthComp@observationGroup <- "Catch"

## ===== SENSITIVITY ANALYSIS =====

## M/K      Linf (cm)  F/M      SPR
sens_results <- data.frame()
for (iLinf in seq_along(Linfseq)) {
  for (iMK in seq_along(MKseq)) {
    lh_temp      <- LifeHistoryObj
    lh_temp@Linf <- Linfseq[iLinf]
    lh_temp@MK   <- MKseq[iMK]
    lh_temp@M    <- MKseq[iMK] * lh_temp@K
    
    runOpt <- DoOptDome.aggregated(lh_temp, FleetPars, lengthComp, 
                                   SizeBins, Lc = 0, mod = "GTG")
    
    cat(sprintf("%-8.1f %-10.0f %-8.3f %-8.3f\n",
                MKseq[iMK], Linfseq[iLinf],
                runOpt$lbPars["F/M"], runOpt$lbPars["SPR"]))
    
    sens_results <- rbind(sens_results, data.frame(
      Linf = Linfseq[iLinf],
      MK   = MKseq[iMK],
      FM   = round(as.numeric(runOpt$lbPars["F/M"]), 3),
      SPR  = round(as.numeric(runOpt$lbPars["SPR"]), 3)
    ))
  }
}
